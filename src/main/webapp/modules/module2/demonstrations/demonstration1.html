Démonstration 1 La première servlet

Exemple avec configuration dans le fichier web.xml

Le lien suivant permet d'accéder la servlet PremiereServletConfigurationWebXML : /modules/module2/PremiereServletConfigurationWebXML

Voici le code de la servlet :

	 	
package fr.eni.javaee.module2;

import java.io.IOException;
import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

/**
 * Servlet implementation class PremiereServletConfigurationWebXML
 */
public class PremiereServletConfigurationWebXML extends HttpServlet {
	private static final long serialVersionUID = 1L;
       
    /**
     * @see HttpServlet#HttpServlet()
     */
    public PremiereServletConfigurationWebXML() {
        super();
       
    }

	/**
	 * @see HttpServlet#doGet(HttpServletRequest request, HttpServletResponse response)
	 */
	protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
		response.getWriter().append("Configuration par web.xml - Served at : ").append(request.getContextPath());
	}

	/**
	 * @see HttpServlet#doPost(HttpServletRequest request, HttpServletResponse response)
	 */
	protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
		doGet(request, response);
	}

}

Voici le paramétrage dans le descripteur de déploiement (web.xml) :

	 	
<servlet>
	<description>Servlet Module 2 Démonstration 1</description>
	<servlet-name>PremiereServletConfigurationWebXML</servlet-name>
	<servlet-class>fr.eni.javaee.module2.PremiereServletConfigurationWebXML</servlet-class>
</servlet>
	<servlet-mapping>
	<servlet-name>PremiereServletConfigurationWebXML</servlet-name>
	<url-pattern>/modules/module2/PremiereServletConfigurationWebXML</url-pattern>
</servlet-mapping>

Exemple avec configuration par annotation

Le lien suivant permet d'accéder la servlet PremiereServletConfigurationAnnotation : /modules/module2/PremiereServletConfigurationAnnotation

Voici le code de la servlet :

	 	
package fr.eni.javaee.module2;

import java.io.IOException;
import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

/**
 * Servlet implementation class PremiereServletConfigurationAnnotation
 */
@WebServlet("/modules/module2/PremiereServletConfigurationAnnotation")
public class PremiereServletConfigurationAnnotation extends HttpServlet {
	private static final long serialVersionUID = 1L;
       
    /**
     * @see HttpServlet#HttpServlet()
     */
    public PremiereServletConfigurationAnnotation() {
        super();
    }

	/**
	 * @see HttpServlet#doGet(HttpServletRequest request, HttpServletResponse response)
	 */
	protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
		response.getWriter().append("Configuration par annotation - Served at : ").append(request.getContextPath());
	}

	/**
	 * @see HttpServlet#doPost(HttpServletRequest request, HttpServletResponse response)
	 */
	protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
		doGet(request, response);
	}

}
