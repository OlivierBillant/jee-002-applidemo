Démonstration 3 La délégation

Servlet => JSP

Le lien suivant pointe vers une servlet délégant l'affichage à la JSP premiereJSPDansWEB-INF.jsp présente dans le répertoire /WEB-INF/modules/module3/demonstrations (Cf. Démonstration 1): /modules/module3/ServletDelegantReponseAUneJSP

Notez que le bouton Retour ne fonctionnera plus car l'URL associée est relative et nous ne sommes plus sur la même URL de base que dans la démonstration 1 (Le segment /demonstrations/ n'est plus présent).

	 	
package fr.eni.javaee.module3;

import java.io.IOException;

import javax.servlet.RequestDispatcher;
import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

/**
 * Servlet implementation class ServletDelegantReponseAUneJSP
 */
@WebServlet("/modules/module3/ServletDelegantReponseAUneJSP")
public class ServletDelegantReponseAUneJSP extends HttpServlet {
	private static final long serialVersionUID = 1L;
       
    /**
     * @see HttpServlet#HttpServlet()
     */
    public ServletDelegantReponseAUneJSP() {
        super();
    }

	/**
	 * @see HttpServlet#doGet(HttpServletRequest request, HttpServletResponse response)
	 */
	protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
		//RequestDispatcher rd = request.getRequestDispatcher("/WEB-INF/modules/module3/demonstrations/premiereJSPDansWEB-INF.jsp");
		//OU
		//RequestDispatcher rd = this.getServletContext().getRequestDispatcher("/modules/module3/demonstrations/secondePremiereJSP");
		//OU
		RequestDispatcher rd = this.getServletContext().getNamedDispatcher("premiereJSPDansWEB_INF");
		rd.forward(request, response);
	}

	/**
	 * @see HttpServlet#doPost(HttpServletRequest request, HttpServletResponse response)
	 */
	protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
		doGet(request, response);
	}

}
